## 快速幂 qpow
 - require: c++98
 - [x] 封装
 - [ ] 测试

```cpp
ll qpow(ll a, ll b, ll mod = mod) {
    ll res = 1;
    a %= mod;
    while (b) {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

ll qpow(ll a, ll b, ll mod = mod) {
    if (!b) 
        return 1;
    ll ns = qpow(a, b >> 1);
    ns = ns * ns % mod;
    if (b & 1) 
        ns = ns * a % mod;
    return ns;
}

ll qmul(ll a, ll b, ll mod = mod) {
    ll res = 0;
    a %= mod;
    while (b) {
        if (b & 1)
            res = (res + a) % mod;
        a = (a + a) % mod;
        b >>= 1;
    }
    return res;
}

ll qmul(ll a, ll b, ll mod = mod) {
    if (!b) 
        return 0;
    ll ns = qmul(a, b >> 1);
    ns = (ns + ns) % mod;
    if (b & 1) 
        ns = (ns + a) % mod;
    return ns;
}
```
