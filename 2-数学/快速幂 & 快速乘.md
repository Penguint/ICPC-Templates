## 快速幂 qpow
 - require: c++98
 - [x] 封装
 - [ ] 测试

```cpp
ll qpow(ll a, ll b, ll mod = mod) {
    ll res = 1;
    a %= mod;
    while (b) {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

ll qmul(ll a, ll b, ll mod = mod) {
    ll res = 0;
    a %= mod;
    while (b) {
        if (b & 1)
            res = (res + a) % mod;
        a = (a + a) % mod;
        b >>= 1;
    }
    return res;
}
```
