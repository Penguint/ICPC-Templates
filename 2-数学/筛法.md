## 筛法

### 埃氏筛

```c++
array<int, N> gpd;
void Eratosthenes(int n) {
    for (int i = 2; i < n; i++)
        if (!gpd[i])
            for (int j = i; j < n; j += i)
                gpd[j] = i;
}
```

### 欧拉筛

```c++
// primes in [1, n]
vector<int> primes, phi, vis;
void euler_sieve(int n) {
    primes.reserve((n + 1) >> 1); // -> n / log(n);
    phi = vector<int>(n + 1);
    vis = vector<int>(n + 1);
    phi[1] = 1;
    for (int i = 2; i <= n; ++i) {
        if (!vis[i]) {
            phi[i] = i - 1;
            primes.emplace_back(i);
        }
        for (int p : primes) {
            if ((ll)i * p >= n + 1) break;
            vis[i * p] = 1;
            if (i % p) phi[i * p] = phi[i] * (p - 1);
            else {
                phi[i * p] = phi[i] * p;
                break;
            }
        }
    }
}
```
