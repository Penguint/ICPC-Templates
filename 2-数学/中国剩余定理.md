## 中国剩余定理 Chinese remainder theorem

n 个同余方程

$$
x \equiv a_i \pmod {m_i}
$$

要求模数两两互质.

```c++
// int inv(int n, int mod);
ll crt(int n, vector<ll> &a, vector<ll> &m) {
    ll ans = 0, M = 1;
    for (int i = 0; i < n; i++) M *= m[i];
    for (int i = 0; i < n; i++) ans = (ans + a[i] * (M / m[i]) % M * inv(M / m[i], m[i]) % M) % M;
    return ans;
}
```

### 扩展中国剩余定理 EXCRT

适用于模数不互质.

```c++
ll excrt(int n, vector<ll> &a, vector<ll> &m) {
    ll M = m[0], A = a[0], x, y;
    for(int i = 1; i < n; i++) {
        ll g = exgcd(M, m[i], x, y);
        if((a[i] - A) % g) return -1;
        x = (a[i] - A) / g * x; x = (x % (m[i] / g) + m[i] / g) % (m[i] / g);
        A = M * x + A;
        M = M / g * m[i]; A %= M;
    }
    return A;
}
```

```c++
// exgcd();
pair<ll, ll> excrt(int n, vector<ll> &a, vector<ll> &m, int nn) // <最小解，[1,nn]解个数>
{
    ll M = m[0], A = a[0], x, y, gcd;
    for (int i = 1; i < n; i++) {
        gcd = exgcd(M, m[i], x, y);
        if ((a[i] - A) % gcd)
            return {-1, 0};
        ll gx = m[i] / gcd;
        x = x * (a[i] - A) / gcd;
        x %= gx;
        A += x * M;
        M = M / gcd * m[i];
        A %= M;
    }
    if (A <= 0) A += M;
    if (nn < A) return {A, 0};
    return {A, (nn - A) / M + 1};
}
```
