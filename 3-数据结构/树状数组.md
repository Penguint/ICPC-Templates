```cpp
template<class T>
struct BIT {
    vector<T> bit;
    BIT(int n) : bit(n + 1) {}
    int lowbit(int x) {
        return x & (-x);
    }
    template<class Fn>
    void run(int i, Fn fun) {
        while (i <= n) {
            fun(bit[i]);
            i += lowbit(i);
        }
    }
    void add(int i, T x) {
        run(i, [&](int &node) { node += x; });
    }
    void sub(int i, T x) {
        run(i, [&](int &node) { node -= x; });  
    }

    template<class Fn>
    void query(int i, Fn fun) {
        while(i > 0) {
            fun(bit[i]);
            i -= lowbit(i);
        }
    }
    T sum(T i) {
        T s = 0;
        query(i, [&](int &node) { s += node;})
        return s;
    }
}
