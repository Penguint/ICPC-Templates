## 树状数组

```cpp
struct bi_t {
    int bit[maxn];
    int n;
    int lowbit(int i) {return i&(-i);}
    int sum(int i) {
        int s=0;
        while (i>0) {
            s+=bit[i];
            i-=lowbit(i);
        }
        return s;
    }
    void add(int i,int x) {
        while (i<=n) {
            bit[i]+=x;
            i+=lowbit(i);
        }
    }
}
```
