### 路径压缩
```cpp
struct UnionFindSet {
    vector<ll> f;
    ll n;

    UnionFindSet(ll n) : f(n) { clear(n); }
    void clear(ll n) {
        this->n = n;
        for (ll i = 0; i < n; i++)
            f[i] = i;
    }
    ll find(ll x) {
        if (f[x] != x)
            f[x] = find(f[x]);
        return f[x];
    }
    void Union(ll x, ll y) { f[find(x)] = find(y); }
};
```
