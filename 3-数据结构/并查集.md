```cpp
using namespace std;

class UnionFindSet {
  public:
    int f[150001];
    int n;
    void clear() {
        for (int i = 0; i < n; i++)
            f[i] = i;
    }
    int find(int x) {
        if (f[x] != x)
            f[x] = find(f[x]);
        return f[x];
    }
    void Union(int x, int y) {
        f[find(x)] = find(y);
    }
} animal;
```
