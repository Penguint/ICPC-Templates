### 路径压缩 / 启发式合并
```cpp
struct UnionFindSet {
    vector<ll> f, num, rank;
    ll n;

    UnionFindSet(ll n) : f(n), num(n), rank(n) { clear(n); }
    void clear(ll n) {
        this->n = n;
        for (ll i = 0; i < n; i++) {
            f[i] = i;
            num[i] = rank[i] = 1;
        }
    }

    ll find(ll x) { return f[x] == x ? x : f[x] = find(f[x]); }

    void Union(ll x, ll y) { f[find(y)] = find(x); }

    ll find(ll x) { return f[x] == x ? x : find(f[x]); } // 不路径压缩

    void Union(ll x, ll y) { // 按深度合并
        x=find(x), y=find(y);
        if(x == y) return;
        if (rank[x] > rank[y]) {
            f[y] = x;
            num[x] += num[y];
        } else {
            f[x] = y;
            num[y] += num[x];
            if (rank[x] == rank[y])
                rank[f_y]++;
        }
    }
};
```
