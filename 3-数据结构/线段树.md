#include <bits/stdc++.h>

struct node_t {
    int lo, hi, add_tag, max_val, min_val;

    void add(int tag) {
        add_tag += tag;
        max_val += tag;
        min_val += tag;
    }

    void pushdown(node_t &lhs, node_t &rhs) {
        if (add_tag == 0)
            return;
        lhs.add(add_tag);
        rhs.add(add_tag);
        add_tag = 0;
    }

    void maintain(node_t &lhs, node_t &rhs) {
        min_val = std::min(lhs.min_val, rhs.min_val);
        max_val = std::max(lhs.max_val, rhs.max_val);
    }
};

struct seg_t {
    std::vector<node_t> seg;
    seg_t(int n) : seg(n + n) {
        int k = 1;
        while (k < n)
            k = k + k;
        int m = n + n - k, s = k - n;
        for (int i = 0; i < m; i++)
            seg[i + k] = {i, i + 1, 0, 0, 0};
        for (int i = m; i < n; i++)
            seg[i + s] = {i, i + 1, 0, 0, 0};
        for (int i = n - 1; i; i--)
            seg[i] = {seg[i + i].lo, seg[i + i + 1].hi, 0, 0, 0};
    }
    template <class Fun> void update(int l, int r, Fun &&fun, int i = 1) {
        if (r <= seg[i].lo || seg[i].hi <= l)
            return;
        if (l <= seg[i].lo && seg[i].hi <= r)
            return fun(seg[i]);
        seg[i].pushdown(seg[i + i], seg[i + i + 1]);
        update(l, r, fun, i + i + 0);
        update(l, r, fun, i + i + 1);
        seg[i].maintain(seg[i + i], seg[i + i + 1]);
    }

    template <class Fun> void query(int l, int r, Fun &&fun, int i = 1) {
        if (r <= seg[i].lo || seg[i].hi <= l)
            return;
        if (l <= seg[i].lo && seg[i].hi <= r)
            return fun(seg[i]);
        seg[i].pushdown(seg[i + i], seg[i + i + 1]);
        query(l, r, fun, i + i + 0);
        query(l, r, fun, i + i + 1);
    }

    node_t &top() { return seg[1]; }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cout.tie(nullptr);
    std::cin.tie(nullptr);

    int n;
    std::string s, t;
    std::cin >> n >> s;
    t.assign(n + 10, 0);
    seg_t seg(n);
    for (int i = 0, p = 0, lst = 0; i < n; i++) {

        if (s[i] != 'R' && s[i] != 'L') {
            if (t[p] == '(') {
                lst--;
                seg.update(p, n, [](node_t &node) { node.add(-1); });
            } else if (t[p] == ')') {
                lst++;
                seg.update(p, n, [](node_t &node) { node.add(1); });
            }
        }

        if (s[i] == '(') {
            lst++;
            t[p] = '(';
            seg.update(p, n, [](node_t &node) { node.add(1); });
        } else if (s[i] == ')') {
            lst--;
            t[p] = ')';
            seg.update(p, n, [](node_t &node) { node.add(-1); });
        } else if (s[i] == 'L') {
            p--;
            if (p < 0)
                p = 0;
        } else if (s[i] == 'R') {
            p++;
        } else {
            t[p] = s[i];
        }
        int ans = -1;
        if (lst == 0 && seg.top().min_val >= 0)
            ans = seg.top().max_val;
        std::cout << ans << " ";
    }
    return 0;
}
